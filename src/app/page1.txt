// ─── Sub-components ────────────────────────────────────────────────────────

function StarRating({ count }: { count: number }) {
  return (
    <div className="flex gap-0.5">
      {Array.from({ length: count }).map((_, i) => (
        <Star key={i} className="h-4 w-4 fill-amber-400 text-amber-400" />
      ))}
    </div>
  );
}

function CityCard({ name, country, images }: { name: string; country: string; images: string[] }) {
  const [idx, setIdx] = useState(0);

  useEffect(() => {
    const t = setInterval(() => setIdx((p) => (p + 1) % images.length), 3500);
    return () => clearInterval(t);
  }, [images.length]);

  return (
    <div className="relative h-72 rounded-2xl overflow-hidden shadow-lg group cursor-pointer card-lift">
      <Image
        src={images[idx]}
        alt={name}
        fill
        className="object-cover transition-transform duration-700 group-hover:scale-105"
      />
      <div className="absolute inset-0 bg-gradient-to-t from-black/70 via-black/20 to-transparent" />
      <div className="absolute bottom-0 left-0 p-5">
        <p className="text-white text-xl font-bold leading-tight">{name}</p>
        <p className="text-white/70 text-sm mt-0.5 flex items-center gap-1">
          <MapPin className="h-3 w-3" /> {country}
        </p>
      </div>
      <div className="absolute top-4 right-4 bg-white/20 backdrop-blur-sm text-white text-xs px-2 py-1 rounded-full flex items-center gap-1 opacity-0 group-hover:opacity-100 transition-opacity">
        <Plane className="h-3 w-3" /> Active routes
      </div>
    </div>
  );
}

function TestimonialsCarousel() {
  const [idx, setIdx] = useState(0);
  const [animating, setAnimating] = useState(false);

  const goTo = useCallback((next: number) => {
    setAnimating(true);
    setTimeout(() => {
      setIdx(next);
      setAnimating(false);
    }, 250);
  }, []);

  useEffect(() => {
    const t = setInterval(() => goTo((idx + 1) % testimonials.length), 5500);
    return () => clearInterval(t);
  }, [idx, goTo]);

  const t = testimonials[idx];

  return (
    <section className="py-24 bg-slate-900 relative overflow-hidden">
      {/* subtle grid pattern */}
      <div
        className="absolute inset-0 opacity-5"
        style={{
          backgroundImage:
            'linear-gradient(rgba(255,255,255,.3) 1px,transparent 1px),linear-gradient(90deg,rgba(255,255,255,.3) 1px,transparent 1px)',
          backgroundSize: '40px 40px',
        }}
      />
      <div className="relative max-w-4xl mx-auto px-4 text-center">
        <p className="text-blue-400 font-semibold text-sm uppercase tracking-widest mb-3">Community Stories</p>
        <h2 className="text-3xl md:text-4xl font-bold text-white mb-12">
          Trusted by travellers worldwide
        </h2>

        <div
          className="bg-white/5 border border-white/10 backdrop-blur-sm rounded-3xl p-8 md:p-12 transition-opacity duration-250"
          style={{ opacity: animating ? 0 : 1 }}
        >
          <Quote className="h-8 w-8 text-blue-400 mx-auto mb-6" />
          <p className="text-lg md:text-xl text-white/90 leading-relaxed mb-8 italic">
            "{t.text}"
          </p>
          <div className="flex flex-col items-center gap-2">
            <StarRating count={t.rating} />
            <p className="font-semibold text-white mt-1">{t.name}</p>
            <p className="text-white/50 text-sm">{t.role}</p>
            <span className="mt-1 text-xs bg-blue-500/20 text-blue-300 border border-blue-500/30 px-3 py-0.5 rounded-full flex items-center gap-1">
              <Plane className="h-3 w-3" /> {t.route}
            </span>
          </div>
        </div>

        <div className="flex justify-center mt-8 gap-2">
          {testimonials.map((_, i) => (
            <button
              key={i}
              onClick={() => goTo(i)}
              className={`h-2 rounded-full transition-all duration-300 ${i === idx ? 'bg-blue-400 w-6' : 'bg-white/20 w-2'}`}
            />
          ))}
        </div>
      </div>
    </section>
  );
}
