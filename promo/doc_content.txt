create table public.intents (  id uuid primary key default uuid_generate_v4(),  type text check (type in ('sender','traveler')),  email text not null,  from_city text,  to_city text,  travel_date date,  price numeric,  status text default 'open', -- open / matched / converted  created_at timestamp default now());

create table public.matches (  id uuid primary key default uuid_generate_v4(),  sender_intent_id uuid references intents(id),  traveler_intent_id uuid references intents(id),  agreed_price numeric,  status text default 'pending', -- pending / agreed / committed / completed / cancelled  both_accepted boolean default false,  created_at timestamp default now());

create table public.match_participants (  id uuid primary key default uuid_generate_v4(),  match_id uuid references matches(id) on delete cascade,  user_id uuid references users(id) on delete cascade,  role text check (role in ('sender','traveler')),  created_at timestamp default now());

💳 7. PAYMENTS (ESCROW)
create table public.payments (  id uuid primary key default uuid_generate_v4(),  match_id uuid references matches(id),  payer_id uuid references users(id),  amount numeric,  insurance_fee numeric,  service_fee numeric,  stripe_payment_intent text,  status text default 'pending', -- pending / held / released / refunded / locked  created_at timestamp default now());
💬 8. MESSAGES (IN-APP CONTROL)
create table public.messages (  id uuid primary key default uuid_generate_v4(),  match_id uuid references matches(id) on delete cascade,  sender_id uuid references users(id),  content text,  is_flagged boolean default false,  created_at timestamp default now());
⚖️ 9. DISPUTES
create table public.disputes (  id uuid primary key default uuid_generate_v4(),  match_id uuid references matches(id),  raised_by uuid references users(id),  reason text,  description text,  status text default 'open', -- open / under_review / resolved / rejected  resolution text,  created_at timestamp default now());
📎 10. DISPUTE EVIDENCE
create table public.dispute_evidence (  id uuid primary key default uuid_generate_v4(),  dispute_id uuid references disputes(id) on delete cascade,  file_url text,  type text,  uploaded_by uuid references users(id),  created_at timestamp default now());

create table public.cancellations (  id uuid primary key default uuid_generate_v4(),  match_id uuid references matches(id),  cancelled_by uuid references users(id),  reason text,  refund_amount numeric,  penalty_amount numeric,  created_at timestamp default now());
